function promisify(fn) {
  return function (...args) {
    return new Promise((resolve, reject) => {
      fn.apply(
        null,
        args.concat((err, data) => {
          if (err != null) {
            reject(err);
          } else {
            resolve(data);
          }
        })
      );
    });
  };
}
  1. 青蛙跳台阶，一次能跳一次或者两次
function jump(n) {
  if (n === 0) return 0;
  if (n === 1) return 1;
  let prev = 1,
    next = 2;
  for (let i = 2; i < n; i++) {
    let tmp = next;
    next = prev + next;
    prev = tmp;
  }
  return next;
}
  1. 变态青蛙跳台阶，一次能跳 1,2,3...n 阶（动态规划）
// f[n] = f[n - 1] + f[n - 2]+...+f[n - i]+...f[0]
// f[n - 1] = f[n -2] + f[n -3]+...+f[0]
// -> f[n] = 2 * f[n - 1]
function jump2(n) {
  const dp = Array.from(new Array(n + 1), () => 0);
  dp[0] = 1;
  dp[1] = 1;
  dp[2] = 2;
  for (let i = 3; i < n + 1; i++) {
    dp[i] = 2 * dp[i - 1];
  }
  return dp[n];
}

               }
             } else if (key === 'title') {

                 if (o[key] !== record[key]) {
                   flag = false;
                   break;
                 }
               }
             } else {
               flag = false;
               break;
             }
           }
           if (flag) {
             result.push(o);
           }
       }


   	return obj;
 	},

         if (order === 'desc') {
           return b[keyname] - a[keyname]

       })

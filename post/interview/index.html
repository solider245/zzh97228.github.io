function deepClone(obj) {
  if (typeof obj !== 'object' || obj === null) {
      return obj
  }
  let newObj = Array.isArray(obj) ? [] : {}
  for (let key in obj) {
      if (obj.hasOwnProperty(key)) {
          let temp = obj[key]
          if (typeof obj[key] === 'object') {
              temp = deepClone(obj[key])
          } 
          newObj = temp
      }
  }
  return newObj
  
}
